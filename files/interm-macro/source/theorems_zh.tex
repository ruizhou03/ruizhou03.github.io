% Theorem System (adapted from adv-micro-psu)
% Boxes: \defn, \thm, \lem, \cor, \prop, \clm, \fact, \asm, \pf, \ex, \sol, \rmk, \rmkb, \state

% Definition
\newtcbtheorem[number within=chapter]{mydefinition}{定义}
{
    enhanced, frame hidden, titlerule=0mm, toptitle=1mm, bottomtitle=1mm,
    fonttitle=\bfseries\large, coltitle=black,
    colbacktitle=green!20!white, colback=green!10!white,
}{defn}

\NewDocumentCommand{\defn}{m+m}{
    \begin{mydefinition}{#1}{}#2\end{mydefinition}
}
\NewDocumentCommand{\defnr}{mm+m}{
    \begin{mydefinition}{#1}{#2}#3\end{mydefinition}
}

% Theorem
\newtcbtheorem[use counter from=mydefinition]{mytheorem}{定理}
{
    enhanced, frame hidden, titlerule=0mm, toptitle=1mm, bottomtitle=1mm,
    fonttitle=\bfseries\large, coltitle=black,
    colbacktitle=cyan!20!white, colback=cyan!10!white,
}{thm}

\NewDocumentCommand{\thm}{m+m}{
    \begin{mytheorem}{#1}{}#2\end{mytheorem}
}
\NewDocumentCommand{\thmr}{mm+m}{
    \begin{mytheorem}{#1}{#2}#3\end{mytheorem}
}

\newenvironment{thmpf}{
	{\noindent{\it \textbf{定理的证明}}}
	\tcolorbox[blanker,breakable,left=5mm,parbox=false,top=8pt,
    before upper={\parindent15pt}, after skip=10pt,
	borderline west={1mm}{0pt}{cyan!20!white}]
}{
    \textcolor{cyan!20!white}{\hbox{}\nobreak\hfill$\blacksquare$}
    \endtcolorbox
}

\NewDocumentCommand{\thmp}{m+m+m}{
    \begin{mytheorem}{#1}{}#2\end{mytheorem}
    \begin{thmpf}#3\end{thmpf}
}

% Example
\newenvironment{example}[1]{%
    \par\vspace{5pt}
    \ifx&#1&\noindent\textbf{例.}
    \else\noindent\textbf{例（#1）.}\fi
    \tcolorbox[blanker,breakable,left=5mm,parbox=false,top=8pt,
    before upper={\parindent15pt}, after skip=10pt,
    borderline west={1mm}{0pt}{cyan!10!white}]
}{\endtcolorbox\vspace{5pt}}

\NewDocumentCommand{\ex}{O{}+m}{
    \begin{example}{#1}#2\end{example}
}

% Solution
\newenvironment{solution}{%
    \par\vspace{5pt}
    \noindent\textbf{解.}
    \tcolorbox[blanker,breakable,left=5mm,parbox=false,
    before upper={\parindent15pt}, after skip=10pt,

}{\endtcolorbox\vspace{5pt}}

\NewDocumentCommand{\sol}{+m}{\begin{solution}#1\end{solution}}
