\textbf{Definition (Inner Product)} An inner product on a vector space
\(V\) is

\[<\cdot,\cdot>:V\times V\to \F\]

that satisfies:

\begin{itemize}
\item
  \(<f,f> \ge 0\) and \(<f,f> = 0\iff f = 0\).
\item
  \(<f+g,h> = <f,h>+<g,h>\).
\item
  \(<f,g> = \overline{<g,f>}\).
\end{itemize}

\textbf{Examples}

\begin{itemize}
\item
  \(<\pr{a_1,\cdots,a_m},\pr{b_1,\cdots,b_m}> = \sum_{n = 1}^Ma_nb_n\).
\item
  \(l^2 = \brace{\pr{a_1,a_2,\cdots}:\sinf \abs{a_k}^2<\infty}\), with
  \(<\pr{a_1,a_2,\cdots},\pr{b_1,b_2,\cdots}> = \sinf a_k\overline{b_k}\).
\item
  \(C\pr{\br{0,1}}\), with \(<f,g> = \int_0^1f\bar g\d x\).
\item
  \(L^2\pr{\mu}\), with \(<f,g> = \int f\bar g\d x\).
\end{itemize}

\emph{\textbf{Proposition}}

\begin{itemize}
\item
  \(<0,f> = \inner{f}{0} = 0\).
\item
  \(\inner{f}{g+h} = \inner{f}{g}+\inner{f}{h}\).
\item
  \(\inner{f}{\alpha g} = \bar\alpha\inner{f}{g}\).
\end{itemize}

Inner product induces a norm.

\textbf{Definition}

\[\norm{f} = \sq{\inner{f}{f}}\]

\textbf{Definition} \(f\) and \(g\) are \emph{orthogonal}, denoted
\(f\perp g\), if

\[\inner{f}{g} = 0\]

\emph{\textbf{Pythagoren Theorem}} If \(f\perp g\), then

\[\norm{f+g}^2 = \norm{f}^2+\norm{g}^2\]

\emph{\textbf{Proof}}

\[\ba
\norm{f+g}^2& =\inner{f+g}{f+g}\\
& = \inner{f}{f}+\inner{f}{g}+\inner{g}{f}+\inner{g}{g}\\
& = \inner{f}{f}+\inner{g}{g}\\
& =\norm{f}^2+\norm{g}^2
\ea\]

\emph{\textbf{Cauchy-S Inequality}}

\[\abs{\inner{f}{g}}\le \norm{f}\norm{g}\]
