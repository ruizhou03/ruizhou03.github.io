\begin{tikzpicture}[scale=1.0]
  \def\A{6.2}\def\Y{7.6}\def\Yz{4.7}
  \pgfmathsetmacro{\mb}{\Yz/\Y}
  \pgfmathsetmacro{\bx}{(\A-\Yz)/(1-\mb)}
  \pgfmathsetmacro{\by}{\A-\bx}
  \pgfmathsetmacro{\rhoB}{\mb*\bx/\by}
  \pgfmathsetmacro{\KB}{\by*pow(\bx,\rhoB)}
  \pgfmathsetmacro{\Ax}{2.5}
  \pgfmathsetmacro{\Ay}{\A-\Ax}
  \pgfmathsetmacro{\rhoA}{\Ax/\Ay}
  \pgfmathsetmacro{\KA}{\Ay*pow(\Ax,\rhoA)}
  \draw[-Latex] (0,0) -- (9.2,0) node[right] {$c_1$};
  \draw[-Latex] (0,0) -- (0,7.4) node[above] {$c_2$};
  \draw[line width=1.4pt] (0,\A) -- (\A,0);     % z=1 budget line (slope -1, intercepts y-g): tangent at A
  \draw[thick] (0,\Yz) -- (\Y,0);               % z>1 budget line (flatter): tangent at B
  \draw[semithick,smooth,samples=120,domain=1.35:5.6] plot (\x,{\KA*pow(\x,-\rhoA)});
  \draw[semithick,smooth,samples=120,domain=1.7:7.4] plot (\x,{\KB*pow(\x,-\rhoB)});
  % equilibria A (z=1) and B (z>1)
  \fill (\Ax,\Ay) circle (1.5pt);
  \node[above right=-1pt and 1pt] at (\Ax,\Ay) {$A$};
  \fill (\bx,\by) circle (1.5pt);
  \node[above right=-1pt and 1pt] at (\bx,\by) {$B$};
  % equilibrium guides for B
  \draw[densely dashed] (0,\by) -- (\bx,\by) -- (\bx,-1.25);
  % guide from A down to its bracket
  \draw[densely dashed] (\Ax,\Ay) -- (\Ax,-2.05);
  % intercept labels
  \node[left] at (0,\A)  {$y-g$};
  \node[left] at (0,\Yz) {$\dfrac{y}{z}$};
  \node[left] at (0,\by) {$c_2^*$};
  \node[below] at (\bx,-0.05) {$c_1^*$};
  \node[below] at (\A,-0.05)  {$y-g$};
  \node[below] at (\Y,-0.05)  {$y$};
  % real money demand brackets: both extend rightward to y
  % z>1 : from B (c1 of B) to y  -> shorter, higher
  \draw (\bx,-0.95) -- (\bx,-1.25) -- (\Y,-1.25) -- (\Y,-0.95);
  \node[below] at ({(\bx+\Y)/2},-1.28) {\small real money demand for $z>1$};
  % z=1 : from A (c1 of A) to y  -> longer, lower
  \draw (\Ax,-1.75) -- (\Ax,-2.05) -- (\Y,-2.05) -- (\Y,-1.75);
  \node[below] at ({(\Ax+\Y)/2},-2.08) {\small real money demand for $z=1$};
\end{tikzpicture}
